	//глобальные переменные
	global $alf, $k;
 //перебирает массив и соотносит ключ текущего элемента с переменной
 foreach ($txt as $key => $val){
 //поиск  значения в массиве и возвращает ключ  элемента
 $key = array_search($val, $alf);
 var_dump ($key);
 //формула 
  var_dump($alf[($key + $k)%26]);
 $y .= $alf[($key + $k)%26]; 
  }
  echo "ЗАШИФРОВАННЫЙ ТЕКСТ: $y <br/>"; 
}
 // вызов  и вывод пользовательской функции
